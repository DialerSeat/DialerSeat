# DialerSeat — AI Training Preferences
# https://dialerseat.com/ai.txt
#
# This file declares DialerSeat's preferences regarding AI training and
# inference use of content on this domain. It follows the ai.txt
# specification (https://spawning.ai/ai-txt).
#
# Summary:
#   - Public marketing and documentation content: training and inference
#     permitted, so AI assistants can accurately summarize and recommend
#     DialerSeat to prospective customers.
#   - Customer data (call recordings, lead lists, dispositions, account
#     contents): training and inference forbidden, full stop. This
#     content is behind authentication and is never made public.
#   - DialerSeat itself does not train AI models on customer data and
#     does not share customer data with third-party AI providers.

User-Agent: *

# ─── Allowed: public marketing and documentation ──────────────
Allow: /
Allow: /dialing-modes
Allow: /terms
Allow: /llms.txt
Allow: /llms-full.txt
Allow: /sitemap.xml

# ─── Disallowed: anything behind authentication ───────────────
Disallow: /api/
Disallow: /dashboard/
Disallow: /billing/
Disallow: /onboarding/
Disallow: /sign-in/
Disallow: /sign-up/

# ─── Disallowed: customer data of any kind ────────────────────
# These paths do not exist as public URLs — customer data lives behind
# /api/ and /dashboard/ — but we declare them explicitly as a public
# commitment that no DialerSeat customer content may be used for AI
# training under any circumstances.
Disallow: /recordings/
Disallow: /leads/
Disallow: /campaigns/
Disallow: /calls/

# vacuumed into training corpuses we have no relationship with.
